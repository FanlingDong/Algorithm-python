find a value in binary tree

```python
def searchBST(root, val):
    if root is None:
        return
    if root.val == val:
        return root
    
    left = searchBST(root.left, val)
    if left is not None:
        return left
    right = searchBST(root.right, val)
    if right is not None:
        return right
    
    else:
        return None
```
